# ลบ remote origin เดิม
git remote remove origin

# เพิ่ม remote origin ใหม่ (แทนที่ YOUR_TOKEN ด้วย Personal Access Token ของคุณ)
git remote add origin https://YOUR_TOKEN@github.com/XXPAK-06/ORDER_IT-KMITL.git

# ตรวจสอบ remote
git remote -v

# ถ้าต้องการ push
git push -u origin main

# รันคำสั่งตามลำดับ

# 1. เพิ่มไฟล์ทั้งหมด
git add .

# 2. Commit การเปลี่ยนแปลง
git commit -m "Update: Add render.yaml and update project structure"

# 3. Push ขึ้น GitHub
git push origin main
